# 143 重排链表

```java
class Solution {
    public void reorderList(ListNode head) {
        if (head == null) return;
        // 找到中间节点
        ListNode cur = head;
        ListNode mid = head;
        while (cur != null && cur.next != null) {
            cur = cur.next.next;
            mid = mid.next;
        }
        // 将右边部分倒序
        ListNode pre = null;
        cur = mid.next;
        mid.next = null;
        while (cur != null) {
            ListNode temp = cur.next;
            cur.next = pre;
            pre = cur;
            cur = temp;
        }
        // 将左边和右边的倒序结合
        ListNode cur1 = head;
        ListNode cur2 = pre;
        while (cur1 != null && cur2 != null) {
            ListNode temp1 = cur1.next;
            ListNode temp2 = cur2.next;
            cur1.next = cur2;
            cur2.next = temp1;
            cur1 = temp1;
            cur2 = temp2;
        }
    }
}
```
