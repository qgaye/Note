# 215 数组中的第K个最大元素

最小堆

```java
class Solution {
    public int findKthLargest(int[] nums, int k) {
        int[] heap = new int[k];
        int i = 0;
        while (i < k) {
            heap[i] = nums[i];
            shiftUp(heap, i);
            i += 1;
        }
        while (i < nums.length) {
            if (heap[0] < nums[i]) {
                heap[0] = nums[i];
                shiftDown(heap, 0);
            }
            i += 1;
        }
        return heap[0];
    }
    private void shiftUp(int[] arr, int i) {
        if (i > 0 && arr[(i - 1) / 2] > arr[i]) {
            swap(arr, i, (i - 1) / 2);
            shiftUp(arr, (i - 1) / 2);
        }
    }

    private void shiftDown(int[] arr, int i) {
        int child1 = i * 2 + 1;
        int child2 = i * 2 + 2;
        int minIndex = i;
        if (child1 < arr.length && arr[child1] < arr[minIndex]) minIndex = child1;
        if (child2 < arr.length && arr[child2] < arr[minIndex]) minIndex = child2;
        if (minIndex != i) {
            swap(arr, i, minIndex);
            shiftDown(arr, minIndex);
        }
    }
    private void swap(int[] arr, int i, int j) {
        int temp = arr[i];
        arr[i] = arr[j];
        arr[j] = temp;
    }
}
```

快排，每次只排一半

```java
class Solution {
    public int findKthLargest(int[] nums, int k) {
        return quickSort(nums, 0, nums.length - 1, k);
    }
    private int quickSort(int[] nums, int i, int j, int k) {
        if (i == j) return nums[i];
        int p = partition(nums, i, j);
        int after = nums.length - p - 1;
        if (k == after + 1) return nums[p];
        else if (k < after + 1) {
            return quickSort(nums, p + 1, j, k);
        } else {
            return quickSort(nums, i, p - 1, k);
        }
    }
    private int partition(int[] nums, int left, int right) {
        int index = left + 1;
        for (int i = left + 1; i <= right; i++) {
            if (nums[i] < nums[left]) {
                swap(nums, i, index);
                index += 1;
            }
        }
        swap(nums, left, index - 1);
        return index - 1;
    }
    private void swap(int[] arr, int i, int j) {
        int temp = arr[i];
        arr[i] = arr[j];
        arr[j] = temp;
    }
}
```
