# 61 旋转链表

```java
class Solution {
    public ListNode rotateRight(ListNode head, int k) {
        if (head == null) return head;
        ListNode cur = head;
        ListNode pre = head;
        for (int i = 0; i < k; i++) {
            if (cur.next == null) cur = head;
            else cur = cur.next;
        }
        while (cur.next != null) {
            cur = cur.next;
            pre = pre.next;
        }
        if (pre.next != null) {
            ListNode ret = pre.next;
            pre.next = null;
            cur.next = head;
            return ret;            
        } else return head;
    }
}
```
